// Create a bullet pattern
function createSpiral(centerX, centerY) {
  const bullets = [];
  for (let i = 0; i < 360; i += 15) {
    const angle = i * 0.0174533;
    const bullet = new Bullet();
    bullet.position.set(centerX, centerY);
    bullet.velocity.set(Math.cos(angle) * 5, Math.sin(angle) * 5);
    bullets.push(bullet);
  }
  return bullets;
}
